CREATE TABLE person  (id int AUTO_INCREMENT, firstName varchar(50) NOT NULL, lastName varchar(50) NOT NULL,
                        mobileNumber varchar(20) NOT NULL,email varchar(50) NOT NULL, city varchar(50) NOT NULL, state varchar(50) NOT NULL,
                        zip long NOT NULL, PRIMARY KEY(id));
